import pygame import random # 1. Initialization pygame.init() # Screen dimensions SCREEN_WIDTH = 800 SCREEN_HEIGHT = 600 screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT)) pygame.display.set_caption("Knight Racing") # Colors WHITE = (255, 255, 255) RED = (200, 0, 0) BLACK = (0, 0, 0) # Game Speed clock = pygame.time.Clock() # 2. Player (Knight) Setup knight_width = 50 knight_height = 60 knight_x = SCREEN_WIDTH // 2 knight_y = SCREEN_HEIGHT - 100 knight_speed = 7 # 3. Obstacles obs_width = 50 obs_height = 50 obs_x = random.randrange(0, SCREEN_WIDTH - obs_width) obs_y = -600 obs_speed = 5 score = 0 # Main Game Loop running = True while running: screen.fill(WHITE) # Clear screen with white for event in pygame.event.get(): if event.type == pygame.QUIT: running = False # 4. Controls keys = pygame.key.get_pressed() if keys[pygame.K_LEFT] and knight_x > 0: knight_x -= knight_speed if keys[pygame.K_RIGHT] and knight_x < SCREEN_WIDTH - knight_width: knight_x += knight_speed # 5. Obstacle Movement obs_y += obs_speed if obs_y > SCREEN_HEIGHT: obs_y = 0 - obs_height obs_x = random.randrange(0, SCREEN_WIDTH - obs_width) score += 1 obs_speed += 0.2 # Difficulty increases over time # 6. Collision Detection knight_rect = pygame.Rect(knight_x, knight_y, knight_width, knight_height) obs_rect = pygame.Rect(obs_x, obs_y, obs_width, obs_height) if knight_rect.colliderect(obs_rect): print(f"Game Over! Your Score: {score}") running = False # Drawing the characters pygame.draw.rect(screen, BLACK, knight_rect) # The Player pygame.draw.rect(screen, RED, obs_rect) # The Obstacle pygame.display.update() clock.tick(60) pygame.quit()
